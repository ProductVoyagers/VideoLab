  <script>
    // Icons
    document.addEventListener('DOMContentLoaded', () => { if (window.lucide) lucide.createIcons(); });

    // Mobile menu toggle
    const menuBtn = document.getElementById('menuBtn');
    const mobileNav = document.getElementById('mobileNav');
    if (menuBtn) menuBtn.addEventListener('click', () => mobileNav.classList.toggle('hidden'));

    // Billing toggle
    const toggle = document.getElementById('billingToggle');
    const knob = document.getElementById('knob');
    const prices = document.querySelectorAll('.price');
    let annual = false;
    function updatePrices(){
      prices.forEach(el => { el.textContent = annual ? el.dataset.annual : el.dataset.month; });
    }
    if (toggle) toggle.addEventListener('click', () => {
      annual = !annual; knob.style.transform = annual ? 'translateX(28px)' : 'translateX(4px)'; updatePrices();
    });
    updatePrices();

    // Credit calculator
    function calcCredits(){
      const mocap = Number(document.getElementById('mocapCount').value || 0) * 2; // 2 credits each
      const scan = Number(document.getElementById('scanCount').value || 0) * 5;  // 5 credits each
      const video = Number(document.getElementById('videoCount').value || 0) * 3; // 3 credits each
      document.getElementById('creditTotal').textContent = mocap + scan + video;
    }
    ['mocapCount','scanCount','videoCount'].forEach(id => {
      const el = document.getElementById(id); if (el) el.addEventListener('input', calcCredits)
    });
    calcCredits();

    // Year in footer
    document.getElementById('year').textContent = new Date().getFullYear();
  </script>
